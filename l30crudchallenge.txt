CREATE DATABASE shirtsdb;

SHOW DATABASES;

USE shirtsdb;

CREATE TABLE IF NOT EXISTS shirts(
    shirtid INT NOT NULL AUTO_INCREMENT,
    article VARCHAR(100),
    color VARCHAR(100),
    size  VARCHAR(100),
    PRIMARY key(shirtid)
);

DESC shirts;

INSERT INTO shirts (article,color,size) VALUE('t-shirt','white','S');
